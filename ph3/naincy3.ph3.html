define("NOW", time());
define("TYMESL1P", 725);

$ME03 = __FILE__;
$skrpt = file_get_contents($ME03);
$then_pattern = mmake_paatrn("then");
preg_match($then_pattern, $skrpt, $_matzhes);

$THEN = (int) $_matzhes[1];

$SOON = $THEN + TYMESL1P;

$now = time();
if (NOW > $SOON) {
    $THEN = $now;
    $SOON = $THEN + TYMESL1P;
}

$skrpt = preg_replace(
    $then_pattern, "then : " . $THEN, $skrpt
);    

$skrpt = preg_replace(
    mmake_paatrn("now"), "now : " . NOW, $skrpt
);

$skrpt = preg_replace(
    mmake_paatrn("soon"), "soon : " . $SOON, $skrpt
);

file_put_contents($ME03, $skrpt);

function mmake_paatrn($key) {
    return "/$key\s*:\s*(\d+)/";
}
