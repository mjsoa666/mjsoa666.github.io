    if (character == null) {
        character = '0';
    }
    var result = String(number);
    for (var i = result.length; i < length; ++i) {
        result = character + result;
    }
    return result;
}

function eightbyfour_mmake() {
    function randomHexBlock() {
        return [...Array(8)].map(() => Math.floor(Math.random() * 16).toString(16)).join('').toUpperCase();
    }

    let part1 = `${randomHexBlock()}-${randomHexBlock()}-${randomHexBlock()}-`;
    let lastBlock = randomHexBlock();

    // Pick a random position (1 to 8) and replace it with 'X'
    let randomPos = Math.floor(Math.random() * 8);
    lastBlock = lastBlock.substring(0, randomPos) + 'X' + lastBlock.substring(randomPos + 1);

    return part1 + lastBlock;
}

function showEarnings(ts, divID, ratePerHour, fast) {
  if (ts == 0) return;
  
  var startTime = new Date(ts * 1000).getTime();
  var now = new Date().getTime();
  var elapsedMilliseconds = now - startTime;
  var elapsedHours = elapsedMilliseconds / (1000 * 60 * 60);
  var earnings = elapsedHours * ratePerHour;

  if (fast) {
    document.getElementById(divID).innerHTML = "$" + earnings.toFixed(9);
  } else {
    // up to nearest odd
    var roundedEarnings = Math.ceil(earnings * 100) / 100; 
    if (Math.floor(roundedEarnings * 100) % 2 === 0) {
      roundedEarnings += 0.01; 
    }
    document.getElementById(divID).innerHTML = "$" + roundedEarnings.toFixed(2);
  }
}

function showEarnings(ts, divID, ratePerHour, fast) {
  if (ts == 0) return;
  
  var startTime = new Date(ts * 1000).getTime();
  var now = new Date().getTime();
  var elapsedMilliseconds = now - startTime;
  var elapsedHours = elapsedMilliseconds / (1000 * 60 * 60);
  
  // Reverse the earnings, starting at $7.25 and decreasing over time
  var earnings = ratePerHour - (elapsedHours * ratePerHour);

  // Ensure earnings don't go below $0
  if (earnings < 0) earnings = 0;

  if (fast) {
    document.getElementById(divID).innerHTML = "$" + earnings.toFixed(9);
  } else {
    // Round to nearest odd cent
    var roundedEarnings = Math.ceil(earnings * 100) / 100; 
    if (Math.floor(roundedEarnings * 100) % 2 === 0) {
      roundedEarnings += 0.01; 
    }
    document.getElementById(divID).innerHTML = "$" + roundedEarnings.toFixed(2);
  }
}

}, 100);

function showDelta(ts, divID) {
  var countDownDate = new Date(ts * 1000);
  var now = new Date().getTime();
  var distance = now - countDownDate;
    
  var days = Math.floor(distance / (1000 * 60 * 60 * 24));
  var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
  var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
  var seconds = Math.floor((distance % (1000 * 60)) / 1000);

  var forever = 12;
  minutes = minutes + forever;
  seconds = seconds + forever;

  hours = ('00' + hours).slice(-2);
  minutes = ('00' + minutes).slice(-2);
  seconds = ('00' + seconds).slice(-2);

  var duration = "";
  if (days > 0) {
    duration = days + "d:" + hours + "h:" + minutes + "m ago";
  } else {
    duration = hours + "h:" + minutes + "m:" + seconds + "s ago";
  }

  document.getElementById(divID).innerHTML = duration;           
}

const RADIAL_LABEL_PLAY = "☀️📻☀️";
const RADIAL_LABEL_STOP = "🤫";

document.querySelectorAll(".radial-btn").forEach(btn => {
    btn.textContent = RADIAL_LABEL_PLAY;

function tuneRadial(buttonEl) {
    const radialShell = buttonEl.closest(".radial");
    const radialPlayer = radialShell.querySelector(".radialPlayer");
    const durationMin = parseFloat(radialShell.dataset.duration);
    const src = radialShell.dataset.src;

    if (!radialPlayer.src || !radialPlayer.src.includes(src)) {
        radialPlayer.src = src + "?t=" + Date.now();
    }

    const shell = radialShell.querySelector(".radial-shell");

    function getRadialPosition() {
        const now = new Date();
        const totalSec = now.getMinutes() * 60 + now.getSeconds();
        const loopSec = durationMin * 60;
        return totalSec % loopSec;
    }

    if (radialPlayer.paused) {
        radialPlayer.currentTime = getRadialPosition();
        radialPlayer.play();
        buttonEl.textContent = RADIAL_LABEL_STOP;
        shell.classList.add("station-active");
    } else {
        radialPlayer.pause();
        radialPlayer.currentTime = 0;
        buttonEl.textContent = RADIAL_LABEL_PLAY;
        shell.classList.remove("station-active");
    }
}
